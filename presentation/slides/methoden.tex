\begin{frame}{$\mathbb{M}$ Visualisieren}
    \begin{center}
        \begin{algorithmic}
            \Function{ist\_Teil\_von\_Mandelbrot}{c}
            \State $i \gets 0$
            \State $iterationen \gets 20$
            \State $z_{n+1}, z_n = 0$
            \While{$i < iterationen$}
            
            \State $z_{n+1} = {z_n}^2 + c$

            \If{$|z_{n+1}| > 2$}
            \State \Return $0$
            \EndIf

            \State $i \gets i + 1$
            \EndWhile

            \State \Return $1$
            \EndFunction
        \end{algorithmic}
    \end{center}
\end{frame}
